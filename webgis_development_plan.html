async function loadLayer(layerId, options) {
    const res  = await fetch(`/api/geojson.php?layer=${layerId}`);
    const data = await res.json();

    const layer = L.geoJSON(data, {
        pointToLayer: (feature, latlng) => {
            return L.marker(latlng, {
                icon: L.divIcon({
                    className: '',
                    html: `<div class="pin" style="background:${options.color}">
                             <i class="fas ${options.icon}"></i>
                           </div>`,
                    iconSize: [30, 30]
                })
            });
        },
        onEachFeature: (feature, layer) => {
            const p = feature.properties;
            let imgs = '';
            if (p.images?.length) {
                imgs = p.images.map(i =>
                    `<img src="${i}" style="width:100%;border-radius:6px;margin-top:6px">`
                ).join('');
            }
            layer.bindPopup(`<b>${p.title}</b><br>${p.description}${imgs}`);
            layer.bindTooltip(p.title);
        }
    });

    return layer;
}
